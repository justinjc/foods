  servings: 2
  ingredients:
    - group: Main
      items:
        - name: whole canned tomatoes
          amount: 28 oz
          instruction: large diced
        - name: red bell pepper
          amount: 1
        - name: onion
          amount: 1
          instruction: diced
        - name: garlic
          amount: 4 cloves
        - name: cilantro
          amount: 1 bunch
        - name: cumin
          amount: 1 tsp
        - name: paprika
          amount: 2 tsp
        - name: aji amarillo pepper paste
          amount: 1 tbsp
        - name: eggs
          amount: 6
  instructions:
    - 'Char red bell pepper directly on stove until completely blackened. Peel skin off. Cover with a bowl for 10 minutes. Dice and reserve.'
    - 'Saute onion until softened. Add garlic, cumin, paprika, and reserved red bell pepper and cook for another few minutes.'
    - 'Add tomatoes and aji amarillo and cook until slightly reduced (around 10 minutes).'
    - 'Make six dents in the tomato mixture and crack an egg in each one.'
    - 'Cook lidded until desired egg consistency.'
    - 'Sprinkle cilantro (and/or parsley) over the top.'
